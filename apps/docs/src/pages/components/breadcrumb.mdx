# Breadcrumb

## Introduction

Indicate the current page’s location within a navigational hierarchy.

  src={require('src/assets/structure/stb_breadcrumb-structure.jpg')}

<ol className="u-textGray u-marginTopTiny">
  <li>
      Breadcrumb items
    <i className="u-text200"> - Indicates pages in hierarchical order. Items are basically button links, except for the current page item.</i>
  </li>
  <li>
      Separator
    <i className="u-text200"> - Icon which is used to separate items.</i>
  </li>
</ol>

## API

export const query = graphql`
  query BreadcrumbQuery {
    Breadcrumb: componentMetadata(displayName: { eq: "Breadcrumb" }) {
      ...ComponentApi_metadata
    }
    BreadcrumbItem: componentMetadata(displayName: { eq: "BreadcrumbItem" }) {
      ...ComponentApi_metadata
    }
  }
`;
